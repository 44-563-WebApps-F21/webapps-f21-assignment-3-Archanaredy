Prim's algorithm

Prim's Algorithm is used to find the minimum spanning tree from a graph. Prim's algorithm finds the subset of edges that includes every vertex of the graph such that the sum of the weights of the edges can be minimized.

Source 

            const int INF = 1000000000;

            struct Edge {
                int w = INF, to = -1;
                bool operator<(Edge const& other) const {
                    return make_pair(w, to) < make_pair(other.w, other.to);
                }
            };
            
            int n;
            vector> adj;
            
            void prim() {
                int total_weight = 0;
                vector min_e(n);
                min_e[0].w = 0;
                set q;
                q.insert({0, 0});
                vector selected(n, false);
                for (int i = 0; i < n; ++i) {
                    if (q.empty()) {
                        cout << "No MST!" << endl;
                        exit(0);
                    }
            
                    int v = q.begin()->to;
                    selected[v] = true;
                    total_weight += q.begin()->w;
                    q.erase(q.begin());
            
                    if (min_e[v].to != -1)
                        cout << v << " " << min_e[v].to << endl;
            
                    for (Edge e : adj[v]) {
                        if (!selected[e.to] && e.w < min_e[e.to].w) {
                            q.erase({min_e[e.to].w, e.to});
                            min_e[e.to] = {e.w, v};
                            q.insert({e.w, e.to});
                        }
                    }
                }
            
                cout << total_weight << endl;
            }
